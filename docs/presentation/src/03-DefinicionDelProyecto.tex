  \begin{itemize}

      \item Mayor control sobre otros elementos del sistema

      \item Facilitar la gestión de aspectos relacionados
      \item Registrar estadísticas de uso de la aplicación
    \end{itemize}
  \end{itemize}
\end{frame}
